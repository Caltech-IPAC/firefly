# create a local user
mysql> CREATE USER '<user_id>'@'localhost' IDENTIFIED BY '<user_psswd>';
mysql> GRANT ALL PRIVILEGES ON `user_management`.* TO '<user_id>'@'localhost';

# allow the user to connect from any host;  you may narrow the hosts by, ie. %.ipac.caltech.edu.
mysql> CREATE USER '<user_id>'@'%' IDENTIFIED BY '<user_psswd>';
mysql> GRANT ALL PRIVILEGES ON `user_management`.* TO '<user_id>'@'%';
